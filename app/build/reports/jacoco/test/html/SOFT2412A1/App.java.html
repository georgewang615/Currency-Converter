    public static void loadBook(String path) {

            book = fh.read();

            System.out.println(e);
        }

    public static void display() {
        newest = book.getIndex(0);
        previous = book.getIndex(1);

        String message = "";
        message += String.format("%-12s", "From/To");
        for (Currency c : newest.getData()) {
            message += String.format("%-12s", c.getName());
        }
        List<List<Double>> newGrid = newest.getRatioGrid();
        List<List<Double>> previousGrid = previous.getRatioGrid();

        List<Currency> currencies = newest.getData();

        for (int i = 0; i < currencies.size(); i++) {
            message += "\n\n";
            message += String.format("%-12s", currencies.get(i).getName());

            for (int j = 0; j < currencies.size(); j++) {
                Double newRatio = newGrid.get(i).get(j);
                Double previousRatio = previousGrid.get(i).get(j);

                if (newRatio.equals(0.0)) {
                    message += String.format("%-12s", "-");
                }

                else {
                    if (newRatio > previousRatio) {
                        message += String.format("%.2f", newRatio);
                        message += String.format("%-8s", " (I)");
                    } else if (newRatio < previousRatio) {
                        message += String.format("%.2f", newRatio);
                        message += String.format("%-8s", " (D)");
                    } else {
                        message += String.format("%-12.2f", newRatio);
                    }
                }
            }
        }
        System.out.println(message);
    }

    static int displayMenu(String header, String[] options, String prompt) {
        System.out.println("\n" + header);

        for (int i = 0; i < options.length; i++) {
            System.out.println((i+1) + ". " + options[i]);
        }

        while (true) {
            int selection = Integer.parseInt(prompt);

            if (selection > 0 && selection <= options.length) {
                return selection;
            } else {
                System.out.println("Invalid menu selection");
            }
        }
    }
